SELECT FI.ID, FN.FISH_NAME, FI.LENGTH
FROM FISH_INFO FI, FISH_NAME_INFO FN
WHERE FI.FISH_TYPE = FN.FISH_TYPE
    AND (FN.FISH_NAME, FI.LENGTH) IN (
        SELECT FN.FISH_NAME, MAX(FI.LENGTH) AS 'LENGTH'
        FROM FISH_INFO FI, FISH_NAME_INFO FN
        WHERE FI.FISH_TYPE = FN.FISH_TYPE
        GROUP BY FN.FISH_NAME
    )
ORDER BY FI.ID;
